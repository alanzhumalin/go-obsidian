package main

import (
	"fmt"
	"strconv"
)

func main() {
	s1 := "123"
	s2 := "-456"
	s3 := "3.14"

	// 1) string -> int
	i, err := strconv.Atoi(s1)
	if err != nil {
		panic(err)
	}

	// 2) string -> int64 (base 10)
	i64, err := strconv.ParseInt(s2, 10, 64)
	if err != nil {
		panic(err)
	}

	// 3) string -> float64
	f64, err := strconv.ParseFloat(s3, 64)
	if err != nil {
		panic(err)
	}

	fmt.Println(i)   // 123
	fmt.Println(i64) // -456
	fmt.Println(f64) // 3.14
}
