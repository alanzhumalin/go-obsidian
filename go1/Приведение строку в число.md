Используем встроенную библиотеку strconv

```
package main

import (
	"fmt"
	"strconv"
)

func main() {
	s := "Привет"

	fmt.Println(len(s))      // 12 (байты)
	fmt.Println([]byte(s))   // UTF-8 байты
	fmt.Println([]rune(s))   // Unicode code points

	num := 3

	s1 := strconv.Itoa(num)
	fmt.Println(s1) // "3"

	s2 := strconv.FormatInt(int64(num), 10)
	fmt.Println(s2) // "3"

	s3 := fmt.Sprintf("%d", num)
	fmt.Println(s3) // "3"
}

```
